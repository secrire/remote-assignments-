function max(numbers){// your code here, for-loop method preferred

let oneMax= numbers[0]; //預設第一個最大
for( let i=0; i< numbers.length; i+=1){
if ( oneMax < numbers[i]) {
oneMax= numbers[i]; //oneMax值可以隨著循環的測試而改變
}
}
return oneMax;
}

document.write( max([1, 2, 4, 5]));
