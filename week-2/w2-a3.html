function avg(data){
// your code here
let productArray = data.products; //朝題目解答，指定對象
let priceSum = 0; //求平均數-> 總和/個數-> 先設總和為變數且值＝0

for (let i=0 ; i< productArray.length ; i+=1){ //總和是連續相加-> loop, 勿忘let
priceSum = productArray[i].price +priceSum;} // ＊＊＊＊＊＊＊priceSum的值會一直隨著循環而變化

let priceAve = priceSum / productArray.length ;
return priceAve;
}

products:[{name:"Product 1",price:100},

]}));
